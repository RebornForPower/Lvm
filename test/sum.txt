   start : OPCLEAR AL     #AL=0
           OPCLEAR CL     #CL=0
           OPLOADVB CL 1  #CL=1
           OPLOADVB AL 1  #AL=1     put 1 to AL
   add:    OPLOAD AL CL   #AL=CL
           OPADDB AL 10   #AL=AL+[10]
           OPSTOREB 10 AL #[10]=AL  
           OPINC CL       #CL=CL+1
           OPCMPVB CL 11  #CMP 11
           OPJG    quit
           OPJNG    add
   quit:   OPPRINTR AL
           OPHALT
